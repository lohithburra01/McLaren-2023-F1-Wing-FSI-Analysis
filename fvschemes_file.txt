{

}

gradSchemes
{
    default         Gauss linear;
}

divSchemes
{
    default         none;
    
    div(tauMC)      Gauss linear;
    div(phi,U)      Gauss limitedLinearV 1;
    div(phi,k)      Gauss limitedLinear 1;

    div(phi,e)      Gauss limitedLinear 1;
    div(phi,Ek)     Gauss limitedLinear 1;

}

}

    reconstruct(U)  vanLeerV;

}

}
